```php
<?php

// 格式化输出到网页上
function dd(...$args) {
    echo '<pre>';
    var_dump(...$args);
    echo '<pre>';
}

// 绝对值
dd(abs(-11)); // int(11)

// 向上取整
dd(ceil(1.1)); // float(2) 注意返回的是float, 转int => (int)ceil(1.1)

// 向下取整
dd(floor(1.8)); // float(1)


// 四舍五入
dd(round(1.3)); // float(1)
dd(round(1.5)); // float(2)

// 最大值
dd(max(1, 6, 7, 8, 2, 3)); // int(8)

// 最小值
dd(min(6, 7, 1, 8, 2, 3)); // int(1)

// 指数
dd(pow(1024, 2)); // int(1048576)

// 获取圆周率(画图可能用到)
dd(M_PI); // float(3.141592653589793)
dd(pi()); // float(3.141592653589793)

// 随机数
dd(rand(1, 10)); // int(5)
dd(mt_rand(1, 10)); // int(9)
dd(random_int(1, 10)); // int(6)
```
